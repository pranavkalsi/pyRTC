def decrease_nice(
    target_nice=-20,
    sudo_pass_file="<saving_location>/.sudo_pass",
    renice_path="/usr/bin/renice"
):
    """
    Try to increase process priority (lower nice value).

    Strategy:
    1) Try psutil directly (works if CAP_SYS_NICE is set)
    2) Fallback to sudo -S renice using a password file
    """

    # Unsupported platforms
    if sys.platform in ("darwin", "win32"):
        logging.info("Nice level adjustment not supported on this platform.")
        return

    pid = os.getpid()

    # ---- Attempt 1: direct psutil ----
    try:
        p = psutil.Process(pid)
        p.nice(target_nice)
        logging.info(f"Process nice set to {target_nice} via psutil.")
        return
    except psutil.AccessDenied:
        pass  # Expected if no CAP_SYS_NICE
    except Exception as e:
        logging.warning(f"psutil nice() failed: {e}")

    # ---- Attempt 2: sudo -S renice ----
    try:
        sudo_pass_file = os.path.expanduser(sudo_pass_file)

        with open(sudo_pass_file, "r") as f:
            subprocess.run(
                ["sudo", "-S", renice_path, str(target_nice), "-p", str(pid)],
                stdin=f,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
                check=True,
            )

        logging.info(f"Process nice set to {target_nice} via sudo renice.")
        return

    except FileNotFoundError:
        logging.warning("renice or sudo not found.")
    except subprocess.CalledProcessError:
        logging.warning("sudo renice failed (wrong password or sudoers config).")
    except Exception as e:
        logging.warning(f"Unable to adjust nice level: {e}")

    logging.warning(
        "Unable to adjust nice level. "
        "Use CAP_SYS_NICE or configure sudoers NOPASSWD."
    )
